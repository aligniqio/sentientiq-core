1. Neural Orbit Loader

// ui/NeuronLoader.tsx
import React from "react";

export default function NeuronLoader() {
  return (
    <div className="relative h-16 w-16">
      {[...Array(6)].map((_, i) => (
        <div
          key={i}
          className="absolute inset-0 animate-spin"
          style={{ animationDuration: `${2 + i * 0.3}s`, transform: `rotate(${i * 60}deg)` }}
        >
          <div className="absolute left-1/2 top-0 h-2 w-2 -translate-x-1/2 rounded-full bg-violet-400 shadow-[0_0_8px_2px_rgba(167,139,250,0.7)]" />

    </div>
  );
}

3. Glowing Consensus Meter

Make the consensus bar pulse like a synapse.

// ui/ConsensusBar.tsx
export function ConsensusBar({ value }: { value: number }) {
  return (
    <div className="flex items-center gap-2">
      <div className="relative h-3 w-40 rounded-full bg-white/10 overflow-hidden">
        <div
          className="absolute inset-y-0 left-0 bg-gradient-to-r from-violet-400 via-sky-400 to-emerald-400 animate-pulse"
          style={{ width: `${value}%` }}
        />
      </div>
      <span className="text-sm font-semibold">{value}%</span>
    </div>
  );
}

5. EVI Pulse Ticker

Little strip at top of EVI page that ticks like a stock board.
// ui/EviTicker.tsx
export function EviTicker({ items }:{ items:{agent:string; evi:number;}[] }) {
  return (
    <div className="overflow-hidden whitespace-nowrap border-b border-white/10 bg-black/40 text-xs font-mono">
      <div className="animate-marquee inline-block py-1">
        {items.map((i,idx)=>(
          <span key={idx} className="mx-6">
            {i.agent}: <span className="font-semibold text-emerald-300">{i.evi.toFixed(1)}</span>
          </span>
        ))}
      </div>
      <style>{`@keyframes marquee{0%{transform:translateX(100%);}100%{transform:translateX(-100%);}}
        .animate-marquee{animation:marquee 20s linear infinite;}`}</style>
    </div>
  );
}
